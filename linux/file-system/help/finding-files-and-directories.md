# Finding files and directories

command find:

{% hint style="info" %}
find . -name filename                         # Search from current directory (.)
{% endhint %}

{% hint style="info" %}
find . -name '\*.txt'                               # will find all .txt files in the current directory
{% endhint %}

command locate:

{% hint style="info" %}
locate filename                                             &#x20;
{% endhint %}
